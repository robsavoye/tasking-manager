backend.api.system.image_upload.SystemImageUploadRestAPI.post (   self) 
Uploads an image using the image upload service
---
tags:
  - system
produces:
  - application/json
parameters:
  - in: header
    name: Authorization
    description: Base64 encoded session token
    required: true
    type: string
    default: Token sessionTokenHere==
  - in: body
    name: body
    required: true
    description: JSON object containing image data that will be uploaded
    schema:
      properties:
          data:
              type: string
              default: base64 encoded image data
          mime:
              type: string
              default: file mime/type
          filename:
              type: string
              default: filename
responses:
  200:
    description: Image uploaded successfully
  400:
    description: Input parameter error
  403:
    description: User is not authorized to upload images
  500:
    description: A problem occurred
  501:
    description: Image upload service not defined
